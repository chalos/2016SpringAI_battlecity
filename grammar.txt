<code>                        ::= <stmt_list>
<NEWLINE>                     ::= \\n
<INDENT>                      ::= \\t
<DEDENT>                      ::= \\b
<stmt_list>                   ::= <stmt> | <stmt> <NEWLINE> <stmt_list>
<stmt>                        ::= <simple_stmt> | <compound_stmt>
<simple_stmt>                 ::= <small_stmt> <NEWLINE>
<small_stmt>                  ::= <expr_stmt> 
<expr_stmt>                   ::= <NAME> <augassign_stmt>
<augassign_stmt>              ::= = <test>

<compound_stmt>               ::= <if_stmt> | <while_stmt> | <if_else_stmt>
<if_stmt>                     ::= if <test> : <suite> 
<if_else_stmt>                ::= if <test> : <suite> else : <suite>
<while_stmt>                  ::= while <test> : <suite>
<suite>                       ::= <simple_stmt> | <NEWLINE> <INDENT> <stmt_list> <DEDENT>

<test>                        ::= <arith_expr> <comp_op> <arith_expr>
<comp_op>                     ::= \\<|\\>|==|\\>=|\\<=|\\<\\>|!=
<arith_expr>                  ::= <term> <add_sub> <term>
<add_sub>                     ::= + | -
<term>                        ::= <factor> <mul_div> <factor>
<mul_div>                     ::= * | /
<factor>                      ::= <add_sub> <factor> | <power>
<power>                       ::= <atom> ** <factor>
<atom>                        ::= <NAME> | <real> | <DIGIT>
<real>                        ::= <int-const>.<int-const>
<int-const>                   ::= <int-const><int-const> | <DIGIT>
<DIGIT>                       ::= 1 | 2 | 3 | 4 | 5 | 6 | 7 | 8 | 9 | 0
<NAME>                        ::= enemies[<enemy_type>][0][<cor>] | players[0][0][<cor>] | <vars> | move_dir | shoot
<enemy_type>                  ::= near_me | near_base | 0 
<cor>                         ::= 0|1
<vars>                        ::= vars[<var_size>]
<var_size>                    ::= <DIGIT>
